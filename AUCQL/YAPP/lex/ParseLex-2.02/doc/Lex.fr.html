            ),

        Parse::Lex->trace;  # Class method
        $lexer = Parse::Lex->new(@token);

        TOKEN:while (1) {
          $token = $lexer->next;
          if (not $lexer->eoi) {
            print "Line $.\t";
            print "Type: ", $token->name, "\t";
            print "Content:->", $token->text, "<-\n";
          } else {
            last TOKEN;
          }

        "a multiline
        string with an embedded "" in it"
